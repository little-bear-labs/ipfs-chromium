144 template <encoding T, typename Traits>
145 std::string basic_algorithm<T, Traits>::encoder::process(
146  std::string_view input) {
147  std::string output;
148  std::size_t isize = input.size();
149  auto partial_blocks = static_cast<float>(input.size()) / input_size();
150  auto num_blocks = static_cast<std::size_t>(partial_blocks);
151  auto osize = static_cast<size_t>(std::ceil(partial_blocks * output_size()));
152  if constexpr (std::is_same_v<typename Traits::execution_style, block_tag>) {
153  num_blocks = static_cast<size_t>(std::ceil(partial_blocks));
154  isize = input_size() * num_blocks;
155  }
156  output.resize(std::max(osize, (output_size() * num_blocks)));
157  auto input_it = std::begin(input);
158  int length = 0;
159  for (std::size_t i = 0; i < isize; ++i, ++input_it) {
160  int carry = i >= input.size() ? 0 : static_cast<unsigned char>(*input_it);
161  int j = 0;
162  for (auto oi = output.rbegin();
163  (oi != output.rend()) && (carry != 0 || j < length); ++oi, ++j) {
164  carry += 256 * (*oi);
165  auto byte = (unsigned char*)(&(*oi));
166  *byte = carry % radix;
167  carry /= radix;
168  }
169  length = j;
170  }
171  std::transform(output.rbegin(), output.rend(), output.rbegin(),
172  [](auto c) { return Traits::charset[c]; });
173  if constexpr (Traits::padding == 0) {
174  output.resize(osize);
175  } else {
176  auto pad_size = output.size() - osize;
177  output.replace(osize, pad_size, pad_size, Traits::padding);
178  }
179  if constexpr (std::is_same_v<typename Traits::execution_style, stream_tag>) {
180  output.erase(0, output.size() % output_size() ? output.size() - length : 0);
181  }
182  return output;
183 }

209 template <encoding T, typename Traits>
210 std::string basic_algorithm<T, Traits>::decoder::process(
211  std::string_view input) {
212  std::string output;
213  auto end = std::find(input.begin(), input.end(), Traits::padding);
214  size_t input_size = std::distance(input.begin(), end);
215  auto partial_blocks = static_cast<float>(input_size) / this->input_size();
216  auto output_size = static_cast<size_t>(this->output_size() * partial_blocks);
217  if constexpr (std::is_same_v<typename Traits::execution_style, block_tag>) {
218  std::size_t num_blocks = 0;
219  auto input_size_float = static_cast<float>(input.size());
220  num_blocks =
221  static_cast<size_t>(std::ceil(input_size_float / this->input_size()));
222  output.resize(this->output_size() * num_blocks);
223  input_size = this->input_size() * num_blocks;
224  } else {
225  output.resize(output_size);
226  }
227  auto input_it = input.begin();
228  for (size_t i = 0; i < input_size; ++i, ++input_it) {
229  int carry = i > input.size() || *input_it == Traits::padding
230  ? 0
231  : valset[(unsigned char)(*input_it)];
232  if (carry == 255) {
233  // throw std::invalid_argument(std::string{"Invalid input character
234  // "} + *input_it);
235  return {};
236  }
237  auto j = output.size();
238  while (carry != 0 || j > 0) {
239  auto index = j - 1;
240  carry += radix * static_cast<unsigned char>(output[index]);
241  output[index] = static_cast<unsigned char>(carry % 256);
242  carry /= 256;
243  if (carry > 0 && index == 0) {
244  output.insert(0, 1, 0);
245  } else {
246  j = index;
247  }
248  }
249  }
250  if constexpr (std::is_same_v<typename Traits::execution_style, block_tag>) {
251  output.erase(output_size, output.size());
252  }
253  return output;
254 }
